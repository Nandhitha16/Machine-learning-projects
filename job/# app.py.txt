# app.py

import streamlit as st
import pandas as pd
import joblib

# Load models and data
df = pd.read_csv("clustered_jobs.csv")
kmeans = joblib.load("kmeans_model.pkl")
vectorizer = joblib.load("vectorizer.pkl")

st.title("Job Clustering and Recommendation System")

user_skills = st.text_input("Enter your skills (comma-separated):")

if user_skills:
    X_new = vectorizer.transform([user_skills])
    cluster = kmeans.predict(X_new)[0]
    st.write(f"Recommended Jobs from Cluster {cluster}")
    matched_jobs = df[df["Cluster"] == cluster]
    st.dataframe(matched_jobs[["Title", "Company", "Location", "Experience", "Skills"]])
